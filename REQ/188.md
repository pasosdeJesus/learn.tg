# R-#188: Allow users to schedule interview for verification in the availability of the verifier and registering it with CalDav

## Objective

Allow users who have completed their profile to request a verification interview. The system shows the verifier's available time slots (based on a CalDAV calendar), lets the user select one, and creates a calendar event in the verifier's CalDAV calendar.

## Motivation

Currently, users can request verification, but there is no formal mechanism to schedule the interview. The verifier (pdJ staff) manually coordinates via WhatsApp/Telegram, which is inefficient and doesn't scale. Additionally, the verifier wants to keep their personal calendar private and not depend on Google Calendar or third-party services like Calendly.

### Components

| Component | Technology | Purpose |
|-----------|------------|---------|
| **Radicale** | CalDAV server | Store calendar events (busy slots, verification appointments) |
| **Calcurse** | CLI client | Verifier manages their calendar from terminal |
| **DAVx⁵** | Android app | Verifier manages calendar from mobile |
| **learn.tg API** | Next.js API routes | Show availability, book appointments |
| **Frontend** | React/Next.js | User selects time slot |

### Calendar Data Flow

1. **Verifier** manages their calendar via Calcurse (terminal) or DAVx⁵ (Android)
2. **Radicale** stores all calendar events
3. **learn.tg** queries Radicale to get busy slots
4. **learn.tg** calculates available slots based on:
   - Working hours (configurable)
   - Busy events (from Radicale)
   - Vacation/blocked days
5. **User** selects an available slot
6. **learn.tg** creates an event in Radicale (marking the slot as busy)

### 5. Component: `VerificationScheduler`

```tsx
// components/VerificationScheduler.tsx
'use client'

import { useState, useEffect } from 'react'
import { useSession } from 'next-auth/react'
import { useAccount } from 'wagmi'
import { useToast } from '@pasosdejesus/m/shadcn-components/ui/use-toast'
import { Button } from '@pasosdejesus/m/shadcn-components/ui/button'
import { createComponentT } from '@/lib/hooks/useTranslation'

interface Slot {
  start: string
  end: string
}

interface Props {
  lang?: string
  onBooked?: () => void
}

export function VerificationScheduler({ lang = 'en', onBooked }: Props) {
  const [slots, setSlots] = useState<Slot[]>([])
  const [selectedSlot, setSelectedSlot] = useState<Slot | null>(null)
  const [isLoading, setIsLoading] = useState(false)
  const [isBooking, setIsBooking] = useState(false)
  const { toast } = useToast()
  const { data: session } = useSession()
  const { address } = useAccount()

  const t = createComponentT(lang, {
    en: {
      title: 'Schedule Verification Interview',
      description: 'Select a time slot for your verification interview. The interview takes about 20-30 minutes.',
      loadingSlots: 'Loading available slots...',
      noSlots: 'No slots available in the next 14 days. Please check back later.',
      book: 'Book Interview',
      booking: 'Booking...',
      success: 'Interview scheduled successfully!',
      error: 'Failed to schedule interview',
      unauthorized: 'Please connect your wallet first',
    },
    es: {
      title: 'Agendar Entrevista de Verificación',
      description: 'Selecciona un horario para tu entrevista de verificación. La entrevista dura entre 20 y 30 minutos.',
      loadingSlots: 'Cargando horarios disponibles...',
      noSlots: 'No hay horarios disponibles en los próximos 14 días. Por favor revisa más tarde.',
      book: 'Agendar Entrevista',
      booking: 'Agendando...',
      success: '¡Entrevista agendada exitosamente!',
      error: 'Error al agendar entrevista',
      unauthorized: 'Por favor conecta tu billetera primero',
    },
  })

  // Fetch available slots on mount
  useEffect(() => {
    if (session?.address && address) {
      fetchSlots()
    }
  }, [session, address])

  const fetchSlots = async () => {
    setIsLoading(true)
    try {
      const res = await fetch('/api/verification/availability?days=14&duration=30')
      const data = await res.json()
      if (data.slots) {
        setSlots(data.slots)
      } else {
        setSlots([])
      }
    } catch {
      toast({ title: t('error'), variant: 'destructive' })
    } finally {
      setIsLoading(false)
    }
  }

  const handleBook = async () => {
    if (!selectedSlot) return
    if (!session?.address || !address) {
      toast({ title: t('unauthorized'), variant: 'destructive' })
      return
    }

    setIsBooking(true)
    try {
      const token = (session.user as any)?.token
      const res = await fetch('/api/verification/book', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          walletAddress: address.toLowerCase(),
          token,
          start: selectedSlot.start,
          end: selectedSlot.end,
        }),
      })
      const data = await res.json()
      if (!res.ok) throw new Error(data.error || t('error'))
      toast({ title: t('success') })
      setSelectedSlot(null)
      onBooked?.()
      fetchSlots() // Refresh slots
    } catch {
      toast({ title: t('error'), variant: 'destructive' })
    } finally {
      setIsBooking(false)
    }
  }

  if (isLoading) {
    return <p className="text-gray-500">{t('loadingSlots')}</p>
  }

  if (slots.length === 0) {
    return <p className="text-gray-500">{t('noSlots')}</p>
  }

  return (
    <div className="space-y-4">
      <p className="text-sm text-gray-600">{t('description')}</p>
      <div className="grid grid-cols-2 md:grid-cols-3 gap-2">
        {slots.map((slot, i) => (
          <Button
            key={i}
            variant={selectedSlot?.start === slot.start ? 'default' : 'outline'}
            className="text-sm"
            onClick={() => setSelectedSlot(slot)}
          >
            {new Date(slot.start).toLocaleString(lang === 'es' ? 'es' : 'en', {
              weekday: 'short',
              month: 'short',
              day: 'numeric',
              hour: '2-digit',
              minute: '2-digit',
            })}
          </Button>
        ))}
      </div>
      {selectedSlot && (
        <Button onClick={handleBook} disabled={isBooking} className="w-full">
          {isBooking ? t('booking') : t('book')}
        </Button>
      )}
    </div>
  )
}
```

### 6. Server Library: `lib/caldav.ts`

```typescript
// lib/caldav.ts
import { CalDAVClient } from 'ts-caldav'

const CALDAV_URL = process.env.CALDAV_URL || ''
const CALDAV_USER = process.env.CALDAV_USER || ''
const CALDAV_PASS = process.env.CALDAV_PASS || ''
const CALDAV_CALENDAR = process.env.CALDAV_CALENDAR || 'default'

let client: CalDAVClient | null = null

function getClient(): CalDAVClient {
  if (!client) {
    client = new CalDAVClient({
      baseURL: CALDAV_URL,
      auth: { username: CALDAV_USER, password: CALDAV_PASS },
    })
  }
  return client
}

export async function getBusySlots(start: Date, end: Date): Promise<{ start: Date; end: Date }[]> {
  const client = getClient()
  const calendars = await client.getCalendars()
  const calendar = calendars.find((c: any) => c.url.includes(CALDAV_CALENDAR)) || calendars[0]
  const events = await client.getEvents({ calendar, startDate: start, endDate: end })
  return events.map((e: any) => ({ start: e.start, end: e.end }))
}

export async function createEvent(summary: string, start: Date, end: Date, description?: string): Promise<void> {
  const client = getClient()
  const calendars = await client.getCalendars()
  const calendar = calendars.find((c: any) => c.url.includes(CALDAV_CALENDAR)) || calendars[0]
  await client.createEvent({
    calendar,
    summary,
    startDate: start,
    endDate: end,
    description,
  })
}
```
